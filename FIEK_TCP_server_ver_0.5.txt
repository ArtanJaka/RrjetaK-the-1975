def Connection(connSocket,addr):
    print("Lidhur me klientin: "+str(addr))
    while 1:
        try:      
            request = connSocket.recv(4096)

            connSocket.send(response)
        except:
            print ("Lidhja me klientin :"+str(addr)+"u nderpre")

            break
